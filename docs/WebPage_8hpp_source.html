111  bool content_type_ok = true;
112  std::string ext (m_get_extension_from_path (m_file_path));

123  m_content_type = "text/css";
124  }
125  else if (ext == "js")

127  m_content_type = "application/javascript";
128  }
129  else if (ext == "wasm")
130  {
131  m_content_type = "application/wasm";
132  }
133  else if (ext == "ico")
134  {
135  m_content_type = "image/x-icon";
136  }
137  else
138  {
139  std::cerr << "Warnings: Unknown file type deceted... disabling"
140  " Content-Type tag in http header..."
141  << std::endl;
142  content_type_ok = false;
143  }
144  if (m_input_file.good ())
145  {
146  std::string tmp;
147  m_http_body_ptr->clear ();
148  while (!m_input_file.eof ())
149  m_http_body_ptr->push_back (m_input_file.get ());
150  m_http_body_ptr->pop_back (); //pop EOF char
151  m_content_length = m_itos (m_http_body_ptr->size ());
152  tmp.clear ();
153  tmp = HTTP_VERSION;
154  tmp += ' ';
155  tmp += m_itos (m_status_code.status_number);
156  tmp += ' ';
157  tmp += m_status_code.status_phrase;
158  tmp += '\r';
159  tmp += '\n';
160  if (content_type_ok)
161  {
162  tmp += "Content-Type: ";
163  tmp += m_content_type;
164  tmp += "; charset=utf-8";
165  tmp += '\r';
166  tmp += '\n';
167  }
168  tmp += "Content-Length: ";
169  tmp += m_content_length;
170  tmp += '\r';
171  tmp += '\n';
172  m_http_header = tmp;
173  m_input_file.clear ();
174  m_input_file.seekg (0);
175  return true;
176  }
177  else

179  std::cerr << "Error: file not good" << std::endl;
