156void
157WebServer::m_cgi_parser (const std::string &http_request)
158{
159 std::stringstream ss (http_request);
160 std::string tmp_s;
161 std::cerr << "INFO: WebServer: m_cgi_parser: Start CGI analysis..."
162 << std::endl;
163 ss >> tmp_s;
164 std::cerr << "INFO: WebServer: m_cgi_parser: HTTP method used: " << tmp_s
165 << std::endl;
166 ss >> tmp_s;
167 std::cerr << "INFO: WebServer: m_cgi_parser: CGI string = " << tmp_s
168 << std::endl;
169 if (tmp_s[0] != '/' || tmp_s[1] != '?')
170 {
171 std::cerr << "WARNING: WebServer: m_cgi_parser: the request do not"
172 << " contain a CGI string" << tmp_s << std::endl;
173 return;
174 }
175 tmp_s.erase (0, 2);
176 for (unsigned int i = 0; i < m_n_input_parameter; ++i)
177 {
178 std::string token = tmp_s.substr (0, tmp_s.find_first_of ("&"));
179 size_t eq_pos = token.find_first_of ("=");
180 std::string p_name = token.substr (0, eq_pos);
181 if (p_name == "steps")
182 {
183 m_internal_html_form_input.steps
184 = stoul (token.substr (eq_pos + 1, token.size ()));
185 tmp_s.erase (0, token.size () + 1);
186 std::cerr
187 << "INFO: WebServer: m_cgi_parser: found parameter {name = \'"
188 << p_name << "\',\t value = " << m_internal_html_form_input.steps
189 << "}" << std::endl;
190 continue;
191 }
192 else if (p_name == "cwt")
193 {
194 m_internal_html_form_input.cwt
195 = stod (token.substr (eq_pos + 1, token.size ()));
196 tmp_s.erase (0, token.size () + 1);
197 std::cerr
198 << "INFO: WebServer: m_cgi_parser: found parameter {name = \'"
199 << p_name << "\',\t value = " << m_internal_html_form_input.cwt
200 << "}" << std::endl;
201 continue;
202 }
203 else if (p_name == "hwt")
204 {
205 m_internal_html_form_input.hwt
206 = stod (token.substr (eq_pos + 1, token.size ()));
207 tmp_s.erase (0, token.size () + 1);
208 std::cerr
209 << "INFO: WebServer: m_cgi_parser: found parameter {name = \'"
210 << p_name << "\',\t value = " << m_internal_html_form_input.hwt
211 << "}" << std::endl;
212 continue;
213 }
214 else if (p_name == "Ray")
215 {
216 m_internal_html_form_input.Ray
217 = stod (token.substr (eq_pos + 1, token.size ()));
218 tmp_s.erase (0, token.size () + 1);
219 std::cerr

475void
476WebServer::serve_output_page (void)
477{
478 while (1)
479 {
480 m_get_executor ().poll ();
481 m_close_unused_connection ();
482 if (!m_is_waiting_list_empty ())
483 {
484 for (size_t i = 0; i < m_get_plugged_connection (); ++i)
485 {
486 if (m_get_connection_by_index (i)
487 .connection_ptr->get_socket ()
488 .remote_endpoint ()
489 .address ()
490 == m_first_user_address)
491 {
492 if (m_first_user_status == OUTPUT)
493 {
494 if (!m_raylib_compiled)
495 {
496 std::cerr << "INFO: WebServer: respond_to_all: "
497 "first user at "
498 << "OUTPUT stage" << std::endl;
499 if (::system ("make raylib") != 0) //Shell operation
500 {
501 std::cerr
502 << "ERROR: WebServer: serve_output_page: "
503 "failed raylib for web compilation. "
504 "To resolve and see output execute 'make "
505 "purge && make && "
506 "./Rayleigh-Benard-Convection-Simulator "
507 "-w8080 -s8000 -o'"
508 << std::endl;
509 ::exit (1);
510 }
511 m_pages.push_back (std::unique_ptr<WebPage> (
512 new WebPage ("cnt/raylib.html")));
513 m_pages.push_back (std::unique_ptr<WebPage> (
514 new WebPage ("cnt/raylib.js")));
515 m_pages.push_back (std::unique_ptr<WebPage> (
516 new WebPage ("cnt/raylib.wasm")));
517
518 m_raylib_compiled = true;
519 }
520 if (m_get_connection_by_index (i)
521 .connection_ptr->is_ready_to_send ()
522 && m_raylib_compiled)
523 {
524 if (m_get_connection_by_index (i)
525 .connection_ptr->is_ready_to_receive ())
526 {
527 m_get_connection_by_index (i)
528 .connection_ptr->receive ();
529 switch (m_extract_raylib_request (
530 m_get_connection_by_index (i)
531 .connection_ptr->unload_data ()
532 .get ()))
533 {
534 case 'h': //html
535 m_get_connection_by_index (i)
536 .connection_ptr->load_data (
537 m_pages.at (5)
538 ->get_http_response ());
539 m_get_connection_by_index (i)
540 .connection_ptr->send ();
541 m_output_pages_sent[0]
542 = true; //To check if all pages have been sent
543 break;
544
545 case 'j': //js
546 m_get_connection_by_index (i)
547 .connection_ptr->load_data (
548 m_pages.at (6)
549 ->get_http_response ());
550 m_get_connection_by_index (i)
551 .connection_ptr->send ();
552 m_output_pages_sent[1] = true;
553 break;
554
555 case 'w': //wasm
556 m_get_connection_by_index (i)
557 .connection_ptr->load_data (
558 m_pages.at (7)
559 ->get_http_response ());
560 m_get_connection_by_index (i)
561 .connection_ptr->send ();
562 m_wasm_req_count += 1;
563 m_output_pages_sent[2] = true;
564 break;
565
566 case 'i': //ico
567 m_get_connection_by_index (i)
568 .connection_ptr->load_data (
569 m_pages.at (4)
570 ->get_http_response ());
571 m_get_connection_by_index (i)
572 .connection_ptr->send ();
573 break;
574
575 default:
576 std::cerr
577 << "WARNING: WebServer: respond_to_all: "
578 << " possible error in the raylib "
579 "request parser"
580 << std::endl;
581 break;
582 }
583 }
584 }
585 }
586 }
587 }
588 }
589 else if ((m_wasm_req_count == 2)
590 && m_output_pages_sent[0] //Implementation detail.
591 && m_output_pages_sent[1] && m_output_pages_sent[2])
592 {
593 return;
594 }
595 }
