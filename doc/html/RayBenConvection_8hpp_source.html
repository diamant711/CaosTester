355 static std::chrono::time_point<std::chrono::high_resolution_clock> old_time;
356 float delta_t;
357
358 ++n_calls;
359 if(n_calls > 1) {
360 delta_t = static_cast<float>(
361 std::chrono::duration_cast<std::chrono::milliseconds>(
362 std::chrono::high_resolution_clock::now() - old_time
363 ).count()
364 ) / 1000;
365 if(delta_t > 0) {
366 simulation_state.total = m_END_CICLE;
367 simulation_state.eta = delta_t * (m_END_CICLE - n_calls);
368 simulation_state.velocity = 1.0/delta_t;
369 for(int i = 0; i < 80; ++i)
370 std::putc(' ', stdout);
371 std::cerr << "\rINFO: RayBenConvection: m_ETA: eta = "
372 << std::setw(4) << std::setprecision(3)
373 << simulation_state.eta << "s\tspeed = "
374 << simulation_state.velocity << "cicle/s\tstep = "
375 << n_calls << "/" << simulation_state.total << "\r";
376 ::fflush(stdout);
377 }
378 }
379 old_time = std::chrono::high_resolution_clock::now();
380 return;

383template <typename Scalar>
384Eigen::Matrix<Scalar,-1,-1> RayBenConvection::m_spdiags(const Eigen::Matrix<Scalar, -1, -1>& B,
385 const Eigen::Matrix<int, -1, 1>& d,
386 size_t m, size_t n) {
387 Eigen::SparseMatrix<Scalar> A(m,n);
388
389 typedef Eigen::Triplet<Scalar> T;
390 std::vector<T> triplets;
391 triplets.reserve(std::min(m,n)*d.size());
392
393 for (int k = 0; k < d.size(); k++) {
394 int i_min = std::max(0, -d(k));
395 int i_max = std::min(m - 1, n - d(k) - 1);
396 int B_im_dx_start = m >= n ? d(k) : 0;
397
398 for (int i = i_min; i <= i_max; i++) {
399 triplets.push_back( T(i, i+d(k), B(B_im_dx_start + i, k)) );
400 }
401 }
402
403 A.setFromTriplets(triplets.begin(), triplets.end());
404
405 return Eigen::Matrix<Scalar, -1, -1>(A);
406}

408template <typename Scalar>
409Eigen::Matrix<Scalar,-1,-1> RayBenConvection::m_kron(const Eigen::Matrix<Scalar, -1, -1>& B,
410 const Eigen::Matrix<Scalar, -1, -1>& d) {
411 Eigen::Matrix<Scalar, -1, -1> A;
412 A.resize(B.rows()*d.rows(), B.cols()*d.cols());
413
414 for (int i = 0; i < B.rows(); i++)
415 for (int j = 0; j < B.cols(); j++)
416 for (int h = 0; h < d.rows(); h++)
417 for (int k = 0; k < d.cols(); k++)
418 A(i*d.rows() + h, j * d.cols() + k) = B(i, j) * d(h, k);
419
420 return Eigen::Matrix<Scalar, -1, -1>(A);
421}

438Eigen::Matrix<int, 1, -1> RayBenConvection::m_my_symamd(Eigen::Matrix<Scalar, -1, -1> &mat) {
439 if (mat.rows() != mat.cols()) {
440 std::cerr << "symamd: Matrice non quadrata!" << std::endl;
441 exit(1);
442 }
443 Eigen::SparseMatrix<Scalar> spmat(mat.sparseView());
444 int A[spmat.nonZeros()];
445 int p[mat.cols()+1];
446 int perm[mat.cols()+1];
447 int stats [COLAMD_STATS];
448 int retcode;
449
450 ::memset(A, 0, (spmat.nonZeros()) * sizeof(int));
451 ::memset(p, 0, (mat.cols()+1) * sizeof(int));
452 ::memset(perm, 0, (mat.cols()+1) * sizeof(int));
453 ::memset(stats, 0, COLAMD_STATS * sizeof(int));
454
455 int A_i = 0, old_A_i = 0, p_i = 1;
456 p[0] = 0;
457 for(int i = 0; i < mat.cols(); ++i) {
458 old_A_i = A_i;
459 for(int k = 0; k < mat.rows(); ++k) {
460 if(mat(k,i) != 0) {
461 A[A_i] = k;
462 A_i++;
463 }
464 }
465 if(old_A_i < A_i){
466 p[p_i] = A_i;
467 p_i++;
468 }
469 }
470 p[mat.cols()] = spmat.nonZeros();
471
472 retcode = symamd(mat.cols(), A, p, perm, (double *) NULL, stats, &calloc, &free);
473
474 if(retcode == 1){
475 Eigen::Matrix<int, 1, -1> permMat;
476 permMat.resize(1, mat.cols());
477 for(int i = 0; i < mat.cols(); ++i)
478 permMat(0, i) = perm[i];
479 return permMat;
480 } else if (retcode == 0) {
481 std::cerr << "symamd: something wrong..." << std::endl;
482 return Eigen::Matrix<int,1,1>(0);
483 } else {
484 std::cerr << "symamd: something really wrong happened!" << std::endl;
485 return Eigen::Matrix<int,1,1>(0);
486 }

489void RayBenConvection::m_luP(Eigen::Matrix<double,-1,-1> &mat,
490 std::vector<Eigen::Matrix<double,-1,-1>> &ret_vett){
491
492 Eigen::SparseMatrix<double> spmat = mat.sparseView();
493 Eigen::PartialPivLU<Eigen::Matrix<double,-1,-1>> LU_dec(spmat);
494
495 Eigen::Matrix<double,-1,-1> LU_mat(LU_dec.matrixLU());
496 Eigen::Matrix<double,-1,-1> L_mat = LU_mat.triangularView<Eigen::UpLoType::UnitLower>();
497 Eigen::Matrix<double,-1,-1> U_mat = LU_mat.triangularView<Eigen::UpLoType::Upper>();
498 Eigen::Matrix<double,-1,-1> PL_mat = LU_dec.permutationP().transpose() * L_mat;
499
500 ret_vett.clear();
501 ret_vett.push_back(PL_mat); //[0]
502 ret_vett.push_back(U_mat); //[1]
503
504 return;
505}

532int RayBenConvection::init(unsigned int END_CICLE, double cold_temp, double hot_temp,
533 double Ray_numb, double Pr_numb, double Re_numb ){
534
535 std::cerr << "INFO: RayBenConvection: init: Initializing simulation..." << std::endl;
536 std::chrono::time_point<std::chrono::high_resolution_clock> init_lenght =
537 std::chrono::high_resolution_clock::now();
538 Eigen::setNbThreads(NPROC);
539
540 int ret_value = 0;
541 m_END_CICLE = END_CICLE;
542 m_TN = cold_temp;
543 m_TS = hot_temp;
544 m_Ra = Ray_numb;
545 m_Pr = Pr_numb;
546 m_Re = Re_numb;
547 m_Pe = m_Re * m_Pr;
548 m_Gr= m_Ra / m_Pr;
549
550 if (END_CICLE > 6000){
551 std::cerr << "ERROR: RayBenConvection: init: The number of steps requested "
552 << END_CICLE << " is too high." << std::endl;
553 return -1;
554 }
555
556 if(m_TS != m_TN)
557 m_To = (m_TS < m_TN) ? m_TS : m_TN;
558 else{
559 std::cout << "Please, remeber to provide two different temperatures.";
560 return -2;
561 }
562
563 if (m_TN <= 0 || m_TN >= 100){
564 std::cout << "Default parameters regard liquid water. Work in 1-99 range.";
565 return -3;
566 }
567
568 if (m_TS >= 100){
569 std::cout << "Default parameters regard liquid water. Work in 1-99 range.";
570 return -7;
571 }
572
573 if (Ray_numb < 50 || Ray_numb > 150 ){
574 std::cout << "Working range for Rayleigh number: 50-150.";
575 return -4;
576 }
577
578 if (Pr_numb < 5 || Pr_numb > 9){
579 std::cout << "Working range for Prandtl number: 5-9.";
580 return -5;
581 }
582
583 if (Re_numb < 70 || Re_numb > 130){
584 std::cout << "Working range for Reynolds number: 70-130.";
585 return -6;
586 }
587
588 if (m_TN < 4){
589 std::cout << "Working with linear parameters. Liquid water "
590 "below of 4 Celsius does not follow linear behaviour." << std::endl;
591 std::cout << "A simulation will be provided but the user is "
592 "advised to change working parameters." << std::endl;
593 ret_value += 0b00010000000000000000000000000000;
594 }
595
596 if (m_TS - m_TN > 10){
597 std::cout << "Warning: with this temperature range the linear"
598 " model validity is not guaranteed." << std::endl;
599 std::cout << "The duration of the simulation will depend on the temperature range,"
600 "so nsteps requested cannot be assured." << std::endl;
601 ret_value += 0b00100000000000000000000000000000;
602 }
603
604 //m_dt correction:
605 m_dt = m_tf / m_nt;
606
607 //initialising calculation maxtrices:
608 m_u.setZero();
609 m_v.setZero();
610 m_p.setZero();
611 m_S.setZero();
612 m_T.setConstant(m_To);
613
614 m_Tstar = m_T;
615 m_ustar = m_u;
616 m_uhalf = m_u;
617 m_uconv = m_u;
618 m_vstar = m_v;
619 m_vhalf = m_v;
620 m_vconv = m_v;
621
622 //IMPLICIT DIFFUSION:-
623 //B.C vector(for u)
624 m_bcu.setZero(),
625 m_bcu.row(0).setConstant(2 * m_UW / std::pow(m_dx,2));
626 m_bcu.row(m_bcu.rows()-1).setConstant(2 * m_UE / std::pow(m_dx,2));
627 m_bcu.col(0).setConstant(m_US / std::pow(m_dy,2));
628 m_bcu.col(m_bcu.cols()-1).setConstant(m_UN / std::pow(m_dy,2));
629
630 //B.Cs at the corners:
631 m_bcu(0, 0) = 2 * m_UW / std::pow(m_dx,2) + m_US / std::pow(m_dy,2);
632 m_bcu(m_bcu.rows()-1, 0) = 2 * m_UE / std::pow(m_dx,2) + m_US / std::pow(m_dy,2);
633 m_bcu(0, (m_ny-2)-1) = 2 * m_UW / std::pow(m_dx,2) + m_UN / std::pow(m_dy,2);
634 m_bcu(m_bcu.rows()-1, m_bcu.cols()-1) = 2 * m_UE / std::pow(m_dx,2) + m_UN / std::pow(m_dy,2);
635 m_bcu = m_dt * m_bcu / m_Re;
636
637 //B.C vector(for v)
638 m_bcv.setZero();
639 m_bcv.row(0).setConstant(2 * m_VW / std::pow(m_dx,2));
640 m_bcv.row(m_bcv.rows()-1).setConstant(2 * m_VE / std::pow(m_dx,2));
641 m_bcv.col(0).setConstant(m_VS / std::pow(m_dy,2));
642 m_bcv.col(m_bcv.cols()-1).setConstant(m_VN / std::pow(m_dy,2));
643
644 //B.Cs at the corners:
645 m_bcv(0, 0) = 2 * m_VW / std::pow(m_dx,2) + m_VS / std::pow(m_dy,2);
646 m_bcv(m_bcv.rows()-1, 0) = 2 * m_VE / std::pow(m_dx,2) + m_VS / std::pow(m_dy,2);
647 m_bcv(0, m_bcv.cols()-1) = 2 * m_VW / std::pow(m_dx,2) + m_VN / std::pow(m_dy,2);
648 m_bcv(m_bcv.rows()-1, m_bcv.cols()-1) = 2 * m_VE / std::pow(m_dx,2) + m_VN / std::pow(m_dy,2);
649 m_bcv = m_dt * m_bcv / m_Re;
650
651 //Initialising central difference operator(for u)
652 m_e.resize(m_nx-1, 1);
653 m_i.resize(m_ny-2, 1);
654 m_e.setOnes();
655 m_i.setOnes();
656
657 m_Ax = m_spdiags(
658 static_cast<Eigen::Matrix<double, -1, -1>>(m_e * Eigen::Matrix<double, 1, 3>(1, -2, 1)),
659 Eigen::Matrix<int, 1, 3> (-1, 0, 1),
660 m_nx-1, m_nx-1
661 );
662
663 m_Ay = m_spdiags(
664 static_cast<Eigen::Matrix<double, -1, -1>>(m_i * Eigen::Matrix<double, 1, 3>(1, -2, 1)),
665 Eigen::Matrix<int, 1, 3> (-1, 0, 1),
666 m_ny-2, m_ny-2
667 );
