120 {
121  m_get_executor ().poll ();
122  if (!m_is_waiting_list_empty ())
123  {
124  if (!m_connected)
125  {
126  m_status = HANDSHAKE_ANSWARE;
127  m_connected = true;
128  }
129  switch (m_status)
130  {
131  case HANDSHAKE_ANSWARE:
132  std::cerr << "INFO: WebSocketServer: respond: waiting for WS request"
133  << std::endl;
134  while (m_get_first_connection ()
135  .connection_ptr->get_socket ()
136  .available ()
137  <= 0)
138  {
139  }
140  std::cerr << "INFO: WebSocketServer: respond: recived WS request, "
141  "responding"
142  << std::endl;
143  m_get_first_connection ().connection_ptr->receive ();
144  m_handshake_response = "HTTP/1.1 101 Switching Protocols\r\n"
145  "Upgrade: websocket\r\n"
146  "Connection: Upgrade\r\n"
147  "Sec-WebSocket-Accept: "
148  + m_handshake_respond_builder (
149  m_get_first_connection ()
150  .connection_ptr->unload_data ()
151  .get ())
152  + "\r\n\r\n";
153  m_get_first_connection ().connection_ptr->load_data (
154  m_handshake_response);
155  m_get_first_connection ().connection_ptr->send ();
156  m_status = UPDATING_CLIENT;
157  break;
158  case UPDATING_CLIENT:
159  if (m_get_first_connection ().connection_ptr->is_ready_to_send ())
160  {
161  m_actual_frame = m_frame_builder ();
162  m_get_first_connection ().connection_ptr->load_data (
163  m_actual_frame);
164  m_get_first_connection ().connection_ptr->send ();
165  }
166  if (m_actual_step >= m_actual_total)
167  {
168  m_status = CLOSING_CONNECTION;
169  }
170  break;
171  case CLOSING_CONNECTION:
172  if (m_get_first_connection ().connection_ptr->is_ready_to_send ())
173  {
174  m_delete_connection_by_index (0);
175  return false;
176  }
177  break;
178  }
179  }
180  return true;

192 ::std::string
193 WebSocketServer::m_base64_encode (const ::std::string &bindata)
194 {
195  using ::std::numeric_limits;
196  using ::std::string;
197 
198  if (bindata.size () > (numeric_limits<string::size_type>::max () / 4u) * 3u)
199  {
200  throw ::std::length_error ("Converting too large a string to base64.");
201  }
202 
203  const ::std::size_t binlen = bindata.size ();
204  // Use = signs so the end is properly padded.
205  string retval ((((binlen + 2) / 3) * 4), '=');
206  ::std::size_t outpos = 0;
207  int bits_collected = 0;
208  unsigned int accumulator = 0;
209  const string::const_iterator binend = bindata.end ();
210 
211  for (string::const_iterator i = bindata.begin (); i != binend; ++i)
212  {
213  accumulator = (accumulator << 8) | (*i & 0xffu);
214  bits_collected += 8;
215  while (bits_collected >= 6)
216  {
217  bits_collected -= 6;
218  retval[outpos++]
219  = m_b64_table[(accumulator >> bits_collected) & 0x3fu];
220  }
221  }
222  if (bits_collected > 0)
223  { // Any trailing bits that are missing.
224  assert (bits_collected < 6);
225  accumulator <<= 6 - bits_collected;
226  retval[outpos++] = m_b64_table[accumulator & 0x3fu];
227  }
228  assert (outpos >= (retval.size () - 2));
229  assert (outpos <= retval.size ());
230  return retval;
231 }

236 ::std::string
237 WebSocketServer::m_base64_decode (const ::std::string &ascdata)
238 {
239  using ::std::string;
240  string retval;
241  const string::const_iterator last = ascdata.end ();
242  int bits_collected = 0;
243  unsigned int accumulator = 0;
244 
245  for (string::const_iterator i = ascdata.begin (); i != last; ++i)
246  {
247  const int c = *i;
248  if (::std::isspace (c) || c == '=')
249  {
250  // Skip whitespace and padding. Be liberal in what you accept.
251  continue;
252  }
253  if ((c > 127) || (c < 0) || (m_reverse_table[c] > 63))
254  {
255  throw ::std::invalid_argument ("This contains characters not legal "
256  "in a base64 encoded string.");
257  }
258  accumulator = (accumulator << 6) | m_reverse_table[c];
259  bits_collected += 6;
260  if (bits_collected >= 8)
261  {
262  bits_collected -= 8;
263  retval
264  += static_cast<char> ((accumulator >> bits_collected) & 0xffu);
265  }
266  }
267  return retval;
268 }

306 WebSocketServer::m_frame_builder (void)
307 {
308  std::vector<unsigned char> frame;
309  std::vector<unsigned char> payload;
310  std::vector<unsigned char> mask;
311  mask.push_back (0xe1);
312  mask.push_back (0xe0);
313  mask.push_back (0x1c);
314  mask.push_back (0xca);
315  /*
316  * FIN = 1 set to 1 implies that every frame is seen as the last, since they are indipendently sent.
317  * RSV1 = 0 using basic WebSocketServer.
318  * RSV2 = 0
319  * RSV3 = 0
320  * OP C = 0 set to 0 implies that our datas will be read as binary.
321  * O = 0
322  * D = 1
323  * E = 0
324  */
325  frame.push_back (0b10000010);
326  /*
327  * MASK = 1
328  * L = 0
329  * E = 0
330  * N = 0
331  * G = 1
332  * H = 1
333  * T = 0
334  * . = 0
335  */
336  frame.push_back (0b10000000 + sizeof (m_actual_eta)
337  + sizeof (m_actual_velocity10) + sizeof (m_actual_total)
338  + sizeof (m_actual_step));
339  /* mask key = 0xe1e01cca */
340  frame.push_back (mask[0]);
341  frame.push_back (mask[1]);
342  frame.push_back (mask[2]);
343  frame.push_back (mask[3]);
344  /* payload eta */
345  payload.push_back ((m_actual_eta & 0xFF000000) >> 24);
346  payload.push_back ((m_actual_eta & 0x00FF0000) >> 16);
347  payload.push_back ((m_actual_eta & 0x0000FF00) >> 8);
348  payload.push_back ((m_actual_eta & 0x000000FF) >> 0);
349  /* payload velocity10 */
350  payload.push_back ((m_actual_velocity10 & 0xFF000000) >> 24);
351  payload.push_back ((m_actual_velocity10 & 0x00FF0000) >> 16);
352  payload.push_back ((m_actual_velocity10 & 0x0000FF00) >> 8);
353  payload.push_back ((m_actual_velocity10 & 0x000000FF) >> 0);
354  /* payload total */
355  payload.push_back ((m_actual_total & 0xFF000000) >> 24);
356  payload.push_back ((m_actual_total & 0x00FF0000) >> 16);
357  payload.push_back ((m_actual_total & 0x0000FF00) >> 8);
358  payload.push_back ((m_actual_total & 0x000000FF) >> 0);
359  /* payload step */
360  payload.push_back ((m_actual_step & 0xFF000000) >> 24);
361  payload.push_back ((m_actual_step & 0x00FF0000) >> 16);
362  payload.push_back ((m_actual_step & 0x0000FF00) >> 8);
363  payload.push_back ((m_actual_step & 0x000000FF) >> 0);
364  for (size_t i = 0; i < payload.size (); ++i)
365  {
366  frame.push_back (payload.at (i) ^ mask[i % 4]);
367  }
368  return frame;
369 }
