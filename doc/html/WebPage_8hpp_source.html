80bool WebPage::m_fill_http_header(void) {
81 bool content_type_ok = true;
82 std::string ext(m_get_extension_from_path(m_file_path));
83 if(ext == "plain") {
84 m_content_type = "text/plain";
85 } else if(ext == "html") {
86 m_content_type = "text/html";
87 } else if(ext == "css") {
88 m_content_type = "text/css";
89 } else if(ext == "js") {
90 m_content_type = "application/javascript";
91 } else if(ext == "wasm") {
92 m_content_type = "application/wasm";
93 } else {
94 std::cerr << "Warnings: Unknown file type deceted... disabling"
95 " Content-Type tag in http header..." << std::endl;
96 content_type_ok = false;
97 }
98 if(m_input_file.good()) {
99 std::string tmp;
100 m_http_body.clear();
101 while(!m_input_file.eof())
102 m_http_body += m_input_file.get();
103 m_content_length = m_itos(m_http_body.size());
104 tmp.clear();
105 tmp = HTTP_VERSION;
106 tmp += ' ';
107 tmp += m_itos(m_status_code.status_number);
108 tmp += ' ';
109 tmp += m_status_code.status_phrase;
110 tmp += '\r';
111 tmp += '\n';
112 if(content_type_ok) {
113 tmp += "Content-Type: ";
114 tmp += m_content_type;
115 tmp += "; charset=utf-8";
116 tmp += '\r';
117 tmp += '\n';
118 }
119 tmp += "Content-Length: ";
120 tmp += m_content_length;
121 tmp += '\r';
122 tmp += '\n';
123 m_http_header = tmp;
124 m_input_file.clear();
125 m_input_file.seekg(0);
126 return true;
127 } else {
128 std::cerr << "Error: file not good" << std::endl;
129 return false;
130 }
131}

144const std::string WebPage::m_get_extension_from_path(const std::string &path) {
145 std::string::size_type start_index = path.find_last_of('.');
146 if (start_index == path.size()) {
147 return "plain";
148 } else {
149 return path.substr(start_index + 1, path.size() - start_index + 1);
150 }
151}
